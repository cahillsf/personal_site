#this implementation uses a list of boolean values to track whether a node has been visited to avoid cycles 
def BFSAlgo(graph, start, visited_list):
    queue = []
    queue.append(start)
    visited[start] = True
    while queue:
        s = queue.pop(0)
        print(s, end = " ")
        for i in graph[s]:
            if visited[i] == False:
                queue.append(i)
                visited[i] = True

def DFSConnectedAlgo(graph, start, visited_list):
    lineCount = 1
    print("{}.".format(lineCount), end = ' ')
    visited_list[start] = True 
    print(start, end = ' ')
    for i in graph[start]:
        if visited[i] == False:
            DFSAlgo(graph, i, visited_list)
            print("\n")
            lineCount += 1
            print("{}.".format(lineCount), end = ' ')

#this cell creates a dictionary with node numbers as keys and a weight total as the corresponding value
#the weight total is calculated from the edges travelled along the Kruskal MST from 0 to the target node


from collections import defaultdict 
results_dict = defaultdict(list)
#distance from 0 to 0 is 0
results_dict[0].append(0)

#this outer for loop iterates over every other node from 1 to 29
for i in range(1, 30):
    totalWeight = 0
    nodesList = kruskalShortest[i]
    iterations = 0
    firstInd = 0
    secondInd = 1
    #this inner while loop iterates over each pair of nodes travelled 
    while(iterations < len(nodesList) - 1):
        nodeOne = nodesList[firstInd]
        nodeTwo = nodesList[secondInd]
        found = False
        #this inner for loop iterates through the list of MST edges to match the pair of nodes with its corresponding edge weight
        for x in range(len(mstEdgesList)):
            if(x < (len(mstEdgesList)) and mstEdgesList[x][0] == nodeOne and mstEdgesList[x][1] == nodeTwo):
                totalWeight += mstEdgesList[x][2]['weight']
                found = True
        if(found == False):
            nodeOne = nodesList[secondInd]
            nodeTwo = nodesList[firstInd]
            for t in range(len(mstEdgesList)):
                if(t < len(mstEdgesList) and mstEdgesList[t][0] == nodeOne and mstEdgesList[t][1] == nodeTwo):
                    totalWeight += mstEdgesList[t][2]['weight']
        iterations += 1
        firstInd += 1
        secondInd += 1
    #weight total is added as the value for corresponding key
    results_dict[i].append(totalWeight)
